# 基本配置

config.js文件配置

module.exports = {
    title: 'VuePress', //网站的标题，默认主题几个地方用到
    description: 'Tingno记编写的VUE教程', //网站的描述
    base: '/', //要部署的根目录，不配置的话，都会到网站根目录下面
    lastUpdated: 'Last Updated ', // 显示最后更新时间
    head: [//配置头信息
        ['link', { rel: 'icon', href: '/imgs/logo.png' }]//配置网站的icon
    ],
    themeConfig: { //主题的配置信息
        lastUpdated: '上次更新',
        searchMaxSuggestions: 10, //搜索数量
        nav: [ //导航栏配置
            { text: '首页 ', link: '/' }, //没有子导航的，显示文字和链接
            {
                text: '文章 ', items: [ //有自导航的，就显示文字和子导航对象，下同
                    { text: 'vuePress ', link: '/site/vuePress ' },
                    { text: 'docker ', link: '/site/docker ' },
                ]
            },
            { text: 'GitHub ', link: 'https://github.com/yuanjg/' }
        ],
    }
}
